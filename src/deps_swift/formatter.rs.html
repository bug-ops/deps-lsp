5/// Returns `true` if `name` matches the `owner/repo` GitHub identifier pattern.
6fn is_valid_owner_repo(name: &str) -> bool {
7    static RE: std::sync::OnceLock<regex::Regex> = std::sync::OnceLock::new();
8    let re = RE.get_or_init(|| {
9        regex::Regex::new(r"^[a-zA-Z0-9._-]+/[a-zA-Z0-9._-]+$").expect("hardcoded regex is valid")
10    });
11    re.is_match(name)
12}
13
14/// Formatter for Swift/SPM ecosystem LSP responses.
15pub struct SwiftFormatter;
16
17impl EcosystemFormatter for SwiftFormatter {
18    fn format_version_for_text_edit(&self, version: &str) -> String {
19        version.to_string()
20    }
21
22    fn package_url(&self, name: &str) -> String {
23        if is_valid_owner_repo(name) {
24            format!("https://github.com/{name}")
25        } else {
26            String::new()
27        }
28    }
29
30    fn normalize_package_name(&self, name: &str) -> String {
31        name.to_lowercase()
32    }
33
34    fn version_satisfies_requirement(&self, version: &str, requirement: &str) -> bool {
35        let Ok(ver) = semver::Version::parse(version) else {
36            return false;
37        };
38        let Ok(req) = semver::VersionReq::parse(requirement) else {
39            return false;
40        };
41        req.matches(&ver)
42    }
43
44    fn yanked_message(&self) -> &'static str {
45        "This version has been yanked"
46    }
47
48    fn yanked_label(&self) -> &'static str {
49        "*(yanked)*"
50    }
51}
